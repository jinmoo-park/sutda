    2. **resolveShowdown()** (private):
       ```typescript
       import { evaluateHand, compareHands } from '@sutda/shared';

       const alivePlayers = this.state.players.filter(p => p.isAlive);
       const hands = alivePlayers.map(p => ({
         player: p,
         hand: evaluateHand(p.cards[0], p.cards[1]),
       }));

       // 최강 패 찾기
       let best = hands[0];
       const tiedPlayers = [hands[0]];
       for (const h of hands.slice(1)) {
         const result = compareHands(best.hand, h.hand);
         if (result === 'b') {
           best = h;
           tiedPlayers.length = 0;
           tiedPlayers.push(h);
         } else if (result === 'tie') {
           tiedPlayers.push(h);
         }
       }

       if (tiedPlayers.length > 1) {
         // 동점 -> 재경기 (D-22~D-24)
         this.state.phase = 'rematch-pending';
         this.state.tiedPlayerIds = tiedPlayers.map(t => t.player.id);
         return;
       }

       // 승자 결정
       this.state.winnerId = best.player.id;
       this.state.phase = 'result';
       ```
