    **뒤집기 핸들러:**
    ```typescript
    const handleFlip = (idx: number) => {
      if (!dealingComplete) return;  // 배분 중 클릭 불가
      setFlippedIndices(prev => {
        const next = new Set(prev);
        next.add(idx);
        // 2장 모두 뒤집으면 콜백 호출
        if (next.size >= 2 && onAllFlipped) {
          onAllFlipped();
        }
        return next;
      });
    };
    ```
