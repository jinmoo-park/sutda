훅 API:
- `play(key: string)` — `localStorage.getItem('sutda_sfx_muted') === 'true'`이면 무시. 아니면 `new Audio(`/sfx/${encodeURIComponent(SFX_MAP[key])}`)` 생성, volume=0.7, play(). 동일 키 재생 시 기존 Audio 인스턴스의 currentTime=0 리셋 후 play()
- `isMuted` / `toggleMute()` — `useState`로 `localStorage.getItem('sutda_sfx_muted') === 'true'` 초기값 관리
- Audio 인스턴스를 `useRef<Map<string, HTMLAudioElement>>`로 캐싱 (매번 new Audio 방지)

```typescript
// result phase 진입 시 1회 실행
useEffect(() => {
  if (gameState.phase !== 'result' || !myPlayerId) return;
  const me = gameState.players.find(p => p.id === myPlayerId);
  if (!me) return;
  const iAmWinner = gameState.winnerId === myPlayerId;
  const myCards = me.cards?.filter(Boolean);

  if (iAmWinner) {
    // 내 handType이 땡 계열인지 판별
    if (myCards && myCards.length >= 2) {
      const result = evaluateHand(myCards as Card[]);
      const isDdaeng = result.handType.includes('ttaeng');
      play(isDdaeng ? 'win-ddaeng' : 'win-normal');
    } else {
      play('win-normal');
    }
  } else {
    // 패배 분기
    const hasDdaengPenalty = gameState.ttaengPayments?.some(t => t.playerId === myPlayerId);
    if (hasDdaengPenalty) {
      play('lose-ddaeng-penalty');  // #14 땡값
    } else if (myCards && myCards.length >= 2) {
      const result = evaluateHand(myCards as Card[]);
      const isDdaeng = result.handType.includes('ttaeng');
      if (isDdaeng) {
        play('lose-ddaeng-but-lost');  // #15 땡이지만 패배
      } else {
        play('lose-normal');  // #13 일반 패배
      }
    } else {
      play('lose-normal');
    }
  }
}, [gameState.phase]);
```
